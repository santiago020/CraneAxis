#include #include #include #include #include "esp_timer.h" #include // ===================================================== // WIFI / MQTT // ===================================================== const char* WIFI_SSID = "aaaa"; const char* WIFI_PASSWORD = "12345678"; const char* MQTT_BROKER = "broker.hivemq.com"; const int MQTT_PORT = 1883; const char* CLIENT_ID = "grua_esp32_001"; const char* TOPIC_PUB = "pendulo/esp32/imu"; // ESP -> WEB const char* TOPIC_CMD = "pendulo/esp32/cmd"; // WEB -> ESP const char* TOPIC_MODE = "pendulo/esp32/mode"; // WEB -> ESP WiFiClient wifiClient; PubSubClient mqtt(wifiClient); // ===================================================== // PINES // ===================================================== #define STEP_PIN 26 #define DIR_PIN 27 #define ENDSTOP 33 #define POT_PIN 35 Adafruit_VL53L0X lox = Adafruit_VL53L0X(); // ===================================================== // CONFIG GENERAL // ===================================================== volatile int currentPeriod = 2000; float tolerancia = 0.3; int minPeriod = 800; int maxPeriod = 4000; float period_obj = 2000; float period_actual = 2000; float slewRate = 50; float offset_distancia = -0.5; // ===================================================== // FILTROS // ===================================================== float distancia_filtrada = 0.0; const float deadband_dist = 0.3; float angulo_filtrado = 0.0; const float deadband_ang = 0.7; // ===================================================== // TIMERS // ===================================================== volatile bool flag20ms = false; esp_timer_handle_t timer20ms; esp_timer_handle_t stepTimer; // ===================================================== // MOTOR // ===================================================== volatile bool motionActive = false; portMUX_TYPE timerMux = portMUX_INITIALIZER_UNLOCKED; // ===================================================== // VARIABLES DE ESTADO // ===================================================== float angulo = 0.0; float distancia_cm = 0.0; float distancia_home = 0.0; float Dcurr = 0.0; // posicion actual absoluta 0..10 float Dobj = 0.0; // referencia absoluta 0..10 bool homing = false; // ===================================================== // FILTRO POT // ===================================================== const int N = 10; int bufferPot[N] = {0}; int indexPot = 0; long sumaPot = 0; // ===================================================== // MODOS DE ENTRADA // ===================================================== enum Mode {IDLE, SQUARE, RAMP, SINE}; Mode currentMode = IDLE; int ciclos = 0; float t_wave = 0.0; // ===================================================== // MODOS DE CONTROL // ===================================================== enum ControlMode {CTRL_PID, CTRL_LGR, CTRL_OSC, CTRL_SERVO}; ControlMode currentControlMode = CTRL_PID; // ===================================================== // GANANCIAS // ===================================================== float Kp_vel_PID = 120.0; float Kp_ang_PID = 8.0; float Kp_vel_LGR = 120.0; float Kp_ang_LGR = 8.0; float Kp_vel_OSC = 120.0; float Kp_ang_OSC = 8.0; float Kp_vel_SERVO = 120.0; float Kp_ang_SERVO = 8.0; // salidas para telemetria float u_pos = 0.0; float u_ang = 0.0; // ===================================================== // TIMER 20 ms // ===================================================== void onTimer(void* arg) { flag20ms = true; } // ===================================================== // TIMER STEP // ===================================================== void onStepTimer(void* arg) { portENTER_CRITICAL_ISR(&timerMux); if (motionActive) { digitalWrite(STEP_PIN, HIGH); ets_delay_us(2); digitalWrite(STEP_PIN, LOW); } portEXIT_CRITICAL_ISR(&timerMux); } // ===================================================== // FILTROS // ===================================================== float aplicarFiltro(float nueva_dist) { if (abs(nueva_dist - distancia_filtrada) > deadband_dist) { distancia_filtrada = nueva_dist; } return distancia_filtrada; } float aplicarFiltroAngulo(float nuevo_ang) { if (abs(nuevo_ang - angulo_filtrado) > deadband_ang) { angulo_filtrado = nuevo_ang; } return angulo_filtrado; } // ===================================================== // WIFI // ===================================================== void conectarWiFi() { WiFi.mode(WIFI_STA); WiFi.begin(WIFI_SSID, WIFI_PASSWORD); while (WiFi.status() != WL_CONNECTED) { delay(500); } } // ===================================================== // MQTT // ===================================================== void reconnectMQTT() { while (!mqtt.connected()) { if (mqtt.connect(CLIENT_ID)) { mqtt.subscribe(TOPIC_CMD); mqtt.subscribe(TOPIC_MODE); } else { delay(1500); } } } // ===================================================== // CONTROL MODE // ===================================================== void procesarModoControl(String modo) { modo.trim(); if (modo == "PID") currentControlMode = CTRL_PID; else if (modo == "LGR") currentControlMode = CTRL_LGR; else if (modo == "OSC") currentControlMode = CTRL_OSC; else if (modo == "SERVO") currentControlMode = CTRL_SERVO; } // ===================================================== // COMANDOS // STEP relativo: // F10 -> avanzar +10 desde posicion actual // B2 -> retroceder -2 desde posicion actual // GUI ya convierte absoluto -> relativo // ===================================================== void procesarComando(String cmd) { cmd.trim(); if (cmd.length() == 0) return; if (cmd == "H") { portENTER_CRITICAL(&timerMux); homing = true; currentMode = IDLE; digitalWrite(DIR_PIN, LOW); motionActive = true; portEXIT_CRITICAL(&timerMux); return; } if (cmd == "Q") { currentMode = SQUARE; ciclos = 0; Dobj = 10.0; return; } if (cmd == "R") { currentMode = RAMP; ciclos = 0; t_wave = 0.0; return; } if (cmd == "S") { currentMode = SINE; ciclos = 0; t_wave = 0.0; return; } if (cmd.startsWith("F") || cmd.startsWith("B")) { float delta = cmd.substring(1).toFloat(); delta = constrain(delta, 0.0, 10.0); portENTER_CRITICAL(&timerMux); if (cmd.startsWith("F")) { Dobj = constrain(Dcurr + delta, 0.0, 10.0); } else { Dobj = constrain(Dcurr - delta, 0.0, 10.0); } currentMode = IDLE; portEXIT_CRITICAL(&timerMux); return; } } // ===================================================== // CALLBACK MQTT // ===================================================== void mqttCallback(char* topic, byte* payload, unsigned int length) { String msg = ""; for (unsigned int i = 0; i < length; i++) { msg += (char)payload[i]; } String top = String(topic); if (top == TOPIC_CMD) { procesarComando(msg); } else if (top == TOPIC_MODE) { procesarModoControl(msg); } } // ===================================================== // PUBLICAR 6 DATOS // ref,pos,err_pos,err_ang,u_pos,u_ang // ===================================================== void publicarDatos(float ref, float pos, float errPos, float errAng, float uPos, float uAng) { String payload = ""; payload += String(ref, 3); payload += ","; payload += String(pos, 3); payload += ","; payload += String(errPos, 3); payload += ","; payload += String(errAng, 3); payload += ","; payload += String(uPos, 3); payload += ","; payload += String(uAng, 3); mqtt.publish(TOPIC_PUB, payload.c_str()); } // ===================================================== // CONTROL // ===================================================== void calcularControl(float ErrD, float ErrAng) { float kpVel = Kp_vel_PID; float kpAng = Kp_ang_PID; switch (currentControlMode) { case CTRL_PID: kpVel = Kp_vel_PID; kpAng = Kp_ang_PID; break; case CTRL_LGR: kpVel = Kp_vel_LGR; kpAng = Kp_ang_LGR; break; case CTRL_OSC: kpVel = Kp_vel_OSC; kpAng = Kp_ang_OSC; break; case CTRL_SERVO: kpVel = Kp_vel_SERVO; kpAng = Kp_ang_SERVO; break; } u_pos = kpVel * abs(ErrD); u_ang = kpAng * ErrAng; float base_period = maxPeriod - u_pos; base_period = constrain(base_period, minPeriod, maxPeriod); period_obj = base_period - u_ang; period_obj = constrain(period_obj, minPeriod, maxPeriod); } // ===================================================== // SETUP // ===================================================== void setup() { Serial.begin(115200); Wire.begin(); pinMode(STEP_PIN, OUTPUT); pinMode(DIR_PIN, OUTPUT); pinMode(ENDSTOP, INPUT_PULLUP); pinMode(POT_PIN, INPUT); digitalWrite(STEP_PIN, LOW); if (!lox.begin()) { while (1); } lox.configSensor(Adafruit_VL53L0X::VL53L0X_SENSE_HIGH_SPEED); for (int i = 0; i < N; i++) { bufferPot[i] = 0; } conectarWiFi(); mqtt.setServer(MQTT_BROKER, MQTT_PORT); mqtt.setCallback(mqttCallback); esp_timer_create_args_t t1 = { .callback = &onTimer, .dispatch_method = ESP_TIMER_TASK, .name = "timer20ms" }; esp_timer_create(&t1, &timer20ms); esp_timer_start_periodic(timer20ms, 20000); esp_timer_create_args_t t2 = { .callback = &onStepTimer, .dispatch_method = ESP_TIMER_TASK, .name = "stepTimer" }; esp_timer_create(&t2, &stepTimer); esp_timer_start_periodic(stepTimer, currentPeriod); } // ===================================================== // LOOP // ===================================================== void loop() { if (!mqtt.connected()) { reconnectMQTT(); } mqtt.loop(); static int contador = 0; if (flag20ms) { flag20ms = false; // ===================== HOMING if (homing) { VL53L0X_RangingMeasurementData_t measure; lox.rangingTest(&measure, false); if (measure.RangeStatus != 4) { float nueva_dist = (measure.RangeMilliMeter / 10.0) + offset_distancia; distancia_cm = aplicarFiltro(nueva_dist); } if (digitalRead(ENDSTOP) == LOW) { portENTER_CRITICAL(&timerMux); motionActive = false; portEXIT_CRITICAL(&timerMux); homing = false; distancia_home = distancia_cm; Dcurr = 0.0; Dobj = 0.0; } publicarDatos(Dobj, Dcurr, Dobj - Dcurr, angulo, u_pos, u_ang); return; } // ===================== ANGULO int potValue = analogRead(POT_PIN); sumaPot -= bufferPot[indexPot]; bufferPot[indexPot] = potValue; sumaPot += potValue; indexPot = (indexPot + 1) % N; float potFiltrado = sumaPot / (float)N; float nuevo_angulo = ((potFiltrado / 4095.0) * 270.0) - 118.0; angulo = aplicarFiltroAngulo(nuevo_angulo); if (abs(angulo) <= 0.5) angulo = 0.0; // ===================== DISTANCIA contador++; if (contador >= 2) { contador = 0; VL53L0X_RangingMeasurementData_t measure; lox.rangingTest(&measure, false); if (measure.RangeStatus != 4) { float nueva_dist = (measure.RangeMilliMeter / 10.0) + offset_distancia; distancia_cm = aplicarFiltro(nueva_dist); } } // ===================== POSICION RELATIVA Dcurr = distancia_home - distancia_cm; Dcurr = constrain(Dcurr, 0.0, 10.0); Dobj = constrain(Dobj, 0.0, 10.0); // ===================== MODOS AUTOMATICOS if (currentMode == SQUARE) { if (abs(Dcurr - Dobj) < tolerancia) { if (Dobj >= 9.9) Dobj = 0.0; else { Dobj = 10.0; ciclos++; } if (ciclos >= 5) { currentMode = IDLE; } } } else if (currentMode == RAMP) { t_wave += 0.02; float period = 4.0; float fase = fmod(t_wave, period); if (fase < period / 2.0) Dobj = (fase / (period / 2.0)) * 10.0; else Dobj = (1.0 - (fase - period / 2.0) / (period / 2.0)) * 10.0; Dobj = constrain(Dobj, 0.0, 10.0); if (t_wave >= period * 5.0) { currentMode = IDLE; } } else if (currentMode == SINE) { t_wave += 0.02; float freq = 0.25; Dobj = 5.0 + 5.0 * sin(2.0 * PI * freq * t_wave); Dobj = constrain(Dobj, 0.0, 10.0); if (t_wave >= (1.0 / freq) * 5.0) { currentMode = IDLE; } } // ===================== ERRORES float ErrD = Dobj - Dcurr; float ErrAng = angulo; // ===================== MOVIMIENTO motionActive = (abs(ErrD) >= tolerancia); if (motionActive) { digitalWrite(DIR_PIN, (ErrD > 0) ? HIGH : LOW); } // ===================== CONTROL calcularControl(ErrD, ErrAng); // ===================== PERFIL SUAVE if (period_actual < period_obj) period_actual += slewRate; else if (period_actual > period_obj) period_actual -= slewRate; if (abs(period_actual - period_obj) < slewRate) { period_actual = period_obj; } currentPeriod = (int)period_actual; esp_timer_stop(stepTimer); esp_timer_start_periodic(stepTimer, currentPeriod); // ===================== TELEMETRIA publicarDatos(Dobj, Dcurr, ErrD, ErrAng, u_pos, u_ang); } }
