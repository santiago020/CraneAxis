#include #include #include #include #include "esp_timer.h" #include // ===================================================== // WIFI / MQTT // ===================================================== const char* WIFI_SSID = "aaaa"; const char* WIFI_PASSWORD = "12345678"; const char* MQTT_BROKER = "broker.hivemq.com"; const int MQTT_PORT = 1883; const char* CLIENT_ID = "grua_esp32_001"; const char* TOPIC_PUB = "pendulo/esp32/imu"; // ESP -> WEB const char* TOPIC_CMD = "pendulo/esp32/cmd"; // WEB -> ESP const char* TOPIC_MODE = "pendulo/esp32/mode"; // WEB -> ESP WiFiClient wifiClient; PubSubClient mqtt(wifiClient); // ===================================================== // PINES // ===================================================== #define STEP_PIN 26 #define DIR_PIN 27 #define ENDSTOP 33 #define POT_PIN 35 Adafruit_VL53L0X lox = Adafruit_VL53L0X(); // ===================================================== // CONFIG GENERAL // ===================================================== const float stepsPerCM = 462.0; volatile int currentPeriod = 2000; float tolerancia = 0.3; int minPeriod = 800; int maxPeriod = 4000; // filtro de periodo float period_obj = 2000; float period_actual = 2000; float slewRate = 50; // offset sensor distancia float offset_distancia = -0.5; // ===================================================== // FILTROS // ===================================================== float distancia_filtrada = 0; const float deadband_dist = 0.3; float angulo_filtrado = 0; const float deadband_ang = 0.7; // ===================================================== // TIMERS // ===================================================== volatile bool flag20ms = false; esp_timer_handle_t timer20ms; esp_timer_handle_t stepTimer; // ===================================================== // MOTOR // ===================================================== volatile bool motionActive = false; portMUX_TYPE timerMux = portMUX_INITIALIZER_UNLOCKED; // ===================================================== // VARIABLES DE ESTADO // ===================================================== float angulo = 0.0; float distancia_cm = 0.0; float distancia_home = 0.0; float Dcurr = 0.0; // posicion actual float Dobj = 0.0; // referencia bool homing = false; // ===================================================== // FILTRO POT // ===================================================== const int N = 10; int bufferPot[N] = {0}; int indexPot = 0; long sumaPot = 0; // ===================================================== // MODOS DE ENTRADA // ===================================================== enum InputMode {IN_IDLE, IN_STEP, IN_SQUARE, IN_RAMP, IN_SINE, IN_HOME}; InputMode currentInputMode = IN_IDLE; int ciclos = 0; float t_wave = 0.0; // ===================================================== // MODOS DE CONTROL // ===================================================== enum ControlMode {CTRL_PID, CTRL_LGR, CTRL_OSC, CTRL_SERVO}; ControlMode currentControlMode = CTRL_PID; // ===================================================== // GANANCIAS CONTROL // ===================================================== // -------- PID (el que ya tienes) -------- float Kp_pos_PID = 120.0; // equivalente a velocidad segun error posicion float Kp_ang_PID = 8.0; // compensacion angular // -------- LGR -------- float Kp_pos_LGR = 100.0; float Kp_ang_LGR = 6.0; // -------- OSC -------- float Kp_pos_OSC = 90.0; float Kp_ang_OSC = 5.0; // -------- SERVO -------- float Kp_pos_SERVO = 110.0; float Kp_ang_SERVO = 7.0; // variables de salida de control float u_pos = 0.0; float u_ang = 0.0; // ===================================================== // TIMER 20 ms // ===================================================== void onTimer(void* arg) { flag20ms = true; } // ===================================================== // TIMER STEP // ===================================================== void onStepTimer(void* arg) { portENTER_CRITICAL_ISR(&timerMux); if (motionActive) { digitalWrite(STEP_PIN, HIGH); ets_delay_us(2); digitalWrite(STEP_PIN, LOW); } portEXIT_CRITICAL_ISR(&timerMux); } // ===================================================== // FILTROS // ===================================================== float aplicarFiltro(float nueva_dist) { if (abs(nueva_dist - distancia_filtrada) > deadband_dist) { distancia_filtrada = nueva_dist; } return distancia_filtrada; } float aplicarFiltroAngulo(float nuevo_ang) { if (abs(nuevo_ang - angulo_filtrado) > deadband_ang) { angulo_filtrado = nuevo_ang; } return angulo_filtrado; } // ===================================================== // WIFI // ===================================================== void conectarWiFi() { WiFi.mode(WIFI_STA); WiFi.begin(WIFI_SSID, WIFI_PASSWORD); while (WiFi.status() != WL_CONNECTED) { delay(500); } } // ===================================================== // PROCESAR COMANDO DE ENTRADA // ===================================================== void procesarComando(String cmd) { cmd.trim(); if (cmd == "H") { currentInputMode = IN_HOME; portENTER_CRITICAL(&timerMux); homing = true; digitalWrite(DIR_PIN, LOW); motionActive = true; portEXIT_CRITICAL(&timerMux); } else if (cmd.startsWith("F")) { // ejemplo: F10 float x = cmd.substring(1).toFloat(); x = constrain(x, 0.0, 10.0); Dobj = x; currentInputMode = IN_STEP; currentMode = IN_IDLE; } else if (cmd.startsWith("B")) { // ejemplo: B5 float x = cmd.substring(1).toFloat(); x = constrain(x, 0.0, 10.0); Dobj = x; currentInputMode = IN_STEP; currentMode = IN_IDLE; } else if (cmd == "Q") { currentInputMode = IN_SQUARE; currentMode = IN_SQUARE; ciclos = 0; Dobj = 10.0; } else if (cmd == "R") { currentInputMode = IN_RAMP; currentMode = IN_RAMP; ciclos = 0; t_wave = 0.0; } else if (cmd == "S") { currentInputMode = IN_SINE; currentMode = IN_SINE; ciclos = 0; t_wave = 0.0; } } // ===================================================== // PROCESAR MODO DE CONTROL // ===================================================== void procesarModoControl(String modo) { modo.trim(); if (modo == "PID") { currentControlMode = CTRL_PID; } else if (modo == "LGR") { currentControlMode = CTRL_LGR; } else if (modo == "OSC") { currentControlMode = CTRL_OSC; } else if (modo == "SERVO") { currentControlMode = CTRL_SERVO; } } // ===================================================== // CALLBACK MQTT // ===================================================== void mqttCallback(char* topic, byte* payload, unsigned int length) { String msg = ""; for (unsigned int i = 0; i < length; i++) { msg += (char)payload[i]; } String top = String(topic); if (top == TOPIC_CMD) { procesarComando(msg); } else if (top == TOPIC_MODE) { procesarModoControl(msg); } } // ===================================================== // MQTT RECONNECT // ===================================================== void reconnectMQTT() { while (!mqtt.connected()) { if (mqtt.connect(CLIENT_ID)) { mqtt.subscribe(TOPIC_CMD); mqtt.subscribe(TOPIC_MODE); } else { delay(1500); } } } // ===================================================== // APLICAR LEY DE CONTROL // ===================================================== void calcularControl(float ErrD, float ErrAng) { switch (currentControlMode) { case CTRL_PID: u_pos = Kp_pos_PID * abs(ErrD); u_ang = Kp_ang_PID * angulo; break; case CTRL_LGR: u_pos = Kp_pos_LGR * abs(ErrD); u_ang = Kp_ang_LGR * angulo; break; case CTRL_OSC: u_pos = Kp_pos_OSC * abs(ErrD); u_ang = Kp_ang_OSC * angulo; break; case CTRL_SERVO: u_pos = Kp_pos_SERVO * abs(ErrD); u_ang = Kp_ang_SERVO * angulo; break; } float base_period = maxPeriod - u_pos; base_period = constrain(base_period, minPeriod, maxPeriod); period_obj = base_period - u_ang; period_obj = constrain(period_obj, minPeriod, maxPeriod); } // ===================================================== // PUBLICAR 6 DATOS // ref,pos,err_pos,err_ang,u_pos,u_ang // ===================================================== void publicarDatos(float ref, float pos, float errPos, float errAng, float uPos, float uAng) { String payload = ""; payload += String(ref, 3); payload += ","; payload += String(pos, 3); payload += ","; payload += String(errPos, 3); payload += ","; payload += String(errAng, 3); payload += ","; payload += String(uPos, 3); payload += ","; payload += String(uAng, 3); mqtt.publish(TOPIC_PUB, payload.c_str()); } // ===================================================== // VARIABLE AUXILIAR para mantener compatibilidad con tu logica // ===================================================== enum Mode {IDLE, SQUARE, RAMP, SINE}; Mode currentMode = IDLE; // ===================================================== // SETUP // ===================================================== void setup() { Serial.begin(115200); Wire.begin(); pinMode(STEP_PIN, OUTPUT); pinMode(DIR_PIN, OUTPUT); pinMode(ENDSTOP, INPUT_PULLUP); pinMode(POT_PIN, INPUT); digitalWrite(STEP_PIN, LOW); if (!lox.begin()) { while (1); } lox.configSensor(Adafruit_VL53L0X::VL53L0X_SENSE_HIGH_SPEED); for (int i = 0; i < N; i++) { bufferPot[i] = 0; } conectarWiFi(); mqtt.setServer(MQTT_BROKER, MQTT_PORT); mqtt.setCallback(mqttCallback); esp_timer_create_args_t t1 = { .callback = &onTimer, .dispatch_method = ESP_TIMER_TASK, .name = "timer20ms" }; esp_timer_create(&t1, &timer20ms); esp_timer_start_periodic(timer20ms, 20000); esp_timer_create_args_t t2 = { .callback = &onStepTimer, .dispatch_method = ESP_TIMER_TASK, .name = "stepTimer" }; esp_timer_create(&t2, &stepTimer); esp_timer_start_periodic(stepTimer, currentPeriod); } // ===================================================== // LOOP // ===================================================== void loop() { if (!mqtt.connected()) { reconnectMQTT(); } mqtt.loop(); static int contador = 0; if (flag20ms) { flag20ms = false; // ===================== HOMING if (homing) { VL53L0X_RangingMeasurementData_t measure; lox.rangingTest(&measure, false); if (measure.RangeStatus != 4) { float nueva_dist = (measure.RangeMilliMeter / 10.0) + offset_distancia; distancia_cm = aplicarFiltro(nueva_dist); } if (digitalRead(ENDSTOP) == LOW) { portENTER_CRITICAL(&timerMux); motionActive = false; portEXIT_CRITICAL(&timerMux); homing = false; distancia_home = distancia_cm; Dobj = 0.0; currentInputMode = IN_IDLE; } publicarDatos(Dobj, Dcurr, Dobj - Dcurr, angulo, u_pos, u_ang); return; } // ===================== ANGULO int potValue = analogRead(POT_PIN); sumaPot -= bufferPot[indexPot]; bufferPot[indexPot] = potValue; sumaPot += potValue; indexPot = (indexPot + 1) % N; float potFiltrado = sumaPot / (float)N; float nuevo_angulo = ((potFiltrado / 4095.0) * 270.0) - 118.0; angulo = aplicarFiltroAngulo(nuevo_angulo); if (abs(angulo) <= 0.5) angulo = 0.0; // ===================== DISTANCIA contador++; if (contador >= 2) { contador = 0; VL53L0X_RangingMeasurementData_t measure; lox.rangingTest(&measure, false); if (measure.RangeStatus != 4) { float nueva_dist = (measure.RangeMilliMeter / 10.0) + offset_distancia; distancia_cm = aplicarFiltro(nueva_dist); } } // ===================== POSICION RELATIVA Dcurr = distancia_home - distancia_cm; // ===================== MODOS DE ENTRADA AUTOMATICA if (currentMode == SQUARE) { if (abs(Dcurr - Dobj) < tolerancia) { if (Dobj == 10.0) Dobj = 0.0; else { Dobj = 10.0; ciclos++; } if (ciclos >= 5) { currentMode = IDLE; currentInputMode = IN_IDLE; } } } else if (currentMode == RAMP) { t_wave += 0.02; float period = 4.0; float fase = fmod(t_wave, period); if (fase < period / 2.0) Dobj = (fase / (period / 2.0)) * 10.0; else Dobj = (1.0 - (fase - period / 2.0) / (period / 2.0)) * 10.0; if (t_wave >= period * 5.0) { currentMode = IDLE; currentInputMode = IN_IDLE; } } else if (currentMode == SINE) { t_wave += 0.02; float freq = 0.25; Dobj = 5.0 + 5.0 * sin(2.0 * PI * freq * t_wave); if (t_wave >= (1.0 / freq) * 5.0) { currentMode = IDLE; currentInputMode = IN_IDLE; } } // ===================== ERRORES float ErrD = Dobj - Dcurr; float ErrAng = angulo; // ===================== ACTIVA MOTOR motionActive = (abs(ErrD) >= tolerancia); if (motionActive) { digitalWrite(DIR_PIN, (ErrD > 0) ? HIGH : LOW); } // ===================== CONTROL SEGUN MODO calcularControl(ErrD, ErrAng); // ===================== PERFIL SUAVE if (period_actual < period_obj) period_actual += slewRate; else if (period_actual > period_obj) period_actual -= slewRate; if (abs(period_actual - period_obj) < slewRate) { period_actual = period_obj; } currentPeriod = (int)period_actual; esp_timer_stop(stepTimer); esp_timer_start_periodic(stepTimer, currentPeriod); // ===================== PUBLICAR 6 VARIABLES publicarDatos(Dobj, Dcurr, ErrD, ErrAng, u_pos, u_ang); } }
